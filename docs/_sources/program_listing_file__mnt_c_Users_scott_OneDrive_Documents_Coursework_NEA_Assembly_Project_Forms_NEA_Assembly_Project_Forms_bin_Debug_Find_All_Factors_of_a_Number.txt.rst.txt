   ;This program uses repeated addition to find all the factors of a number
   MOV R1, #10 ; Finds factors of this number
   MOV R2, #2 ; this will iterate through all possible factors (starts at 2 as 1 will be always be a factor)
   LOOP:

   NOTFOUND:
