   ;Uses iterative addition to multiply two numbers together
   MOV R1, #0 ;
   MOV R2, #8 ; What to multiply
   LOOP:
       ADD R3, R3, R2
       ADD R1, R1, #1
       CMP R1, #5 ; How many times to multiply
   BLT LOOP
   OUT R3
