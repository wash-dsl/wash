;This program uses repeated addition to find all the factors of a number
MOV R1, #10 ; Finds factors of this number
MOV R2, #2 ; this will iterate through all possible factors (starts at 2 as 1 will be always be a factor)
LOOP:
MOV R3, R1 ;Assigns R3 to R1
INNERLOOP:
SUB R3, R3, R2 ;Repeatedly takes away R2 from R3 until <= 0
CMP R3, #0
BGT INNERLOOP ;If R3 greater than 0 keep on looping
BLT NOTFOUND ;If less than 0, is not a factor
OUT R2 ;Is factor, so output R2
NOTFOUND:
ADD R2, R2, #1 ;Repeats for every number up to R1
CMP R2, R1
BNE LOOP
HALT
